### Opgave 1
$$
\begin{align}
\int^{4}_{0} 2xy \, dy & = [4xy^{2}]_{0}^{4}  \\
 & = 4x \cdot 4^{2} - 4x \cdot 0^{2} \\
 & = 64x
\end{align}
$$
$$
\begin{align}
\int^{2}_{1} 64x \, dx & = [128x^{2}]_{1}^{2}  \\
 & = 128 \cdot 2^{2} - 128 \cdot 1^{2}  \\
 & = 128 \cdot 4 - 128 \\
 & = 384
\end{align}
$$

### Opgave 2
$$
\begin{align}
\int^{1}_{-1} (x - y) \, dy & = [xy - 2y^{2}]_{-1}^{1} \\
 & = x \cdot 1 - 2 \cdot 1^{2} - (x \cdot (-1) - 2 \cdot (-1)^{2}) \\
 & = x - 1 + x + 2 \\
 & = 2x - 1
\end{align}
$$
$$
\begin{align}
\int^{2}_{0} 2x - 1 \, dx & = [4x^{2} - x]_{0}^{2}  \\
 & = 4 \cdot 2^{2} - 2 - (4 \cdot 0^{2} - 0)  \\
 & = 16 - 2 \\
 & = 14
\end{align}
$$

### Opgave 10
$$
\begin{align}
\int^{2}_{1} xye^{x} \, dy & = [2xe^{x}y^{2}]_{1}^{2} \\
 & = 2xe^{x} \cdot 2^{2} - 2xe^{x} \cdot 1^{2} \\
 & = 8xe^{x} - 2xe^{x} \\
 & = 6xe^{x}
\end{align}
$$
$$
\begin{align}
\int^{1}_{0} 6xe^{x} \, dx & = [6xe^{x}]_{0}^{1} \\
 & = 6 \cdot 1 \cdot e^{1} - 6 \cdot 0 \cdot e^{0} \\
 & = 6e
\end{align}
$$

$$
\iint_{R} \frac{\sqrt{ x }}{y^{2}} \, dA = \int^{4}_{0} \int^{2}_{1} \frac{\sqrt{ x }}{y^{2}} \, dy  \, dx  
$$
$$
\begin{align}
\int^{2}_{1} \sqrt{ x } \cdot y^{-2} \, dy & = \left[ -\sqrt{ x } \cdot \frac{1}{y} \right]_{1}^{2} \\
 & = - \sqrt{ x } \cdot \frac{1}{2} - \left( - \sqrt{ x } \cdot \frac{1}{1} \right) \\
 & = \sqrt{ x } - \frac{1}{2} \sqrt{ x } \\
 & = \frac{1}{2} \sqrt{ x }
\end{align}
$$
$$
\begin{align}
\int^{4}_{0} \frac{1}{2} x^{\frac{1}{2}} \, dx & = \left[ \frac{1}{3}x^{\frac{3}{2}} \right]_{0}^{4} \\
 & = \frac{1}{3} \cdot 4^{\frac{3}{2}} - \frac{1}{3} \cdot 0^{\frac{3}{2}} \\
 & = \frac{1}{3} \cdot 4 \cdot \sqrt{ 4 } \\
 & = \frac{8}{3}
\end{align}
$$

### Opgave 19
$$
\begin{align}
\int^{2x}_{0} xy^{3} \, dy & = \frac{x}{4} \cdot [y^{4}]_{0}^{2x} \\
 & = \frac{x}{4} \cdot ((2x)^{4} - (0)^{4}) \\
 & = 4x^{5}
\end{align}
$$
$$
\begin{align}
\int^{2}_{1} 4x^{5} \, dx & = \frac{4}{6} \cdot [x^{6}]_{1}^{2} \\
 & = \frac{2}{3} \cdot (2^{6} - 1^{6}) \\
 & = \frac{2}{3} \cdot (64 - 1) \\
 & = \frac{126}{3} \\
 & = 42
\end{align}
$$

### Opgave 20
$$
\begin{align}
\int^{2y}_{y} y \, dx & = y \int^{2y}_{y} 1 \, dx  \\
 & = y \cdot [x]_{y}^{2y} \\
 & = y \cdot (2y - y) \\
 & = y^{2}
\end{align}
$$
$$
\begin{align}
\int^{3}_{1} y^{2} \, dy & = \left[ \frac{y^{3}}{3} \right]_{1}^{3} \\
 & = \frac{3^{3}}{3} - \frac{1^{3}}{3} \\
 & = 3^{2} - \frac{1}{3} \\
 & = \frac{26}{3}
\end{align}
$$
